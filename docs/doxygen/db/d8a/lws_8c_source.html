81 static error_t
82 lws_mount_add(struct lws_http_mount** mounts, const struct lws_http_mount* item)
83 {
84  assert(mounts != NULL);
85 
86  int counter = 0;
87  struct lws_http_mount* target = *mounts;
88 
89  if (target == NULL) {
90  target = (struct lws_http_mount*)malloc(sizeof(struct lws_http_mount));
91  if (target == NULL) {
92  log_fatal("malloc() failed!\n");
93  return error_alloc;
94  }
95  } else {
96  do {
97  counter++;
98  } while ((target + counter - 1)->mount_next != NULL);
99 
100  struct lws_http_mount* new = (struct lws_http_mount*)realloc(
101  target, sizeof(struct lws_http_mount) * (counter + 1));
102 
103  if (new == NULL) {
104  log_fatal("realloc() failed!\n");
105  return error_alloc;
106  } else {
107  target = new;
108  (target + counter - 1)->mount_next = (target + counter);
109  }
110  }
111 
112  memcpy((target + counter), item, sizeof(struct lws_http_mount));
113  *mounts = target;
114 
115  return error_none;
116 }

145  if (new == NULL) {

151  memset(target + counter + 1, 0, sizeof(struct lws_protocols));
152  }
153 
154  memcpy((target + counter), item, sizeof(struct lws_protocols));
155  *protocols = target;
156 
157  return error_none;

160 static int
161 lws_functions_init(lws_params_t* params, lws_t* lws)
162 {
163  assert(params != NULL);
164  assert(lws != NULL);
165  assert(lws->mounts == NULL);
166  assert(lws->protocols == NULL);
167 
168  error_t e;
169 
170  if (1 /*files enabled*/) {
171  const struct lws_http_mount files = {
172  .mountpoint = "/",
173  .mountpoint_len = 1,
174  .origin_protocol = LWSMPRO_FILE,
175  .origin = params->network_file_dir,
176  .def = "index.html",
177  };
178 
179  e = lws_mount_add(&lws->mounts, &files);
180  if (ERR_NONE != e.err) {
181  log_fatal("lws_mount_add(files) failed!\n");
182  goto fail;
183  }
184  }
185 
186  const lws_plugin_t plugins[] = {
187  { .name = "api", .init_cb = api_lws_init_cb, .user = params->api },
188  { .name = "jpeg", .init_cb = jpeg_lws_init_cb, .user = params->jpeg },
189  //{ .name = "log", .init_cb = log_ram_lws_init_cb, .user = NULL },
190  { 0 }
191  };
192 
193  const lws_plugin_t* plugin = plugins;
194 
195  while (NULL != plugin->name) {
196  assert(NULL != plugin->init_cb);
197 
198  struct lws_http_mount mount;
199  struct lws_protocols protocol;
200 
201  e = plugin->init_cb(plugin->user, &mount, &protocol);
202  switch (e.err) {
203  case ERR_NONE:
204  e = lws_mount_add(&lws->mounts, &mount);
205  if (ERR_NONE != e.err) {
206  log_fatal("lws_mount_add(%s) failed!\n", plugin->name);
207  goto fail;
208  }
209 
210  e = lws_protocol_add(&lws->protocols, &protocol);
211  if (ERR_NONE != e.err) {
212  log_fatal("lws_protocol_add(%s) failed!\n", plugin->name);
213  goto fail;
214  }
215  break;
216  case ERR_NOTBUILTIN:
217  log_trace("lws plugin '%s' not built-in.\n", plugin->name);
218  break;
219  case ERR_NOTINITED:
220  log_trace("lws plugin '%s' can`t be intited.\n", plugin->name);
221  break;
222  default:
223  assert(0);
224  break;
225  }
226 
227  plugin++;
228  }
229 
230  return 0;
231 
232 fail:
233  free(lws->mounts);
234  free(lws->protocols);
235 
236  return 1;
237 }
