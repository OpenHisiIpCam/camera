57 stdout_callback(log_Event* ev)

59  char buf[16];
60  buf[strftime(buf, sizeof(buf), "%H:%M:%S", ev->time)] = '\0';

176 void
177 log_log(int level, const char* file, int line, const char* fmt, ...)
178 {
179  log_Event ev = {
180  .fmt = fmt,
181  .file = file,
182  .line = line,
183  .level = level,
184  };
185 
186  lock();
187 
188  if (!L.quiet && level >= L.level) {
189  init_event(&ev, stderr);
190  va_start(ev.ap, fmt);
191  stdout_callback(&ev);
192  va_end(ev.ap);
193  }
194 
195 #ifdef MAX_CALLBACKS
196  for (int i = 0; i < MAX_CALLBACKS && L.callbacks[i].fn; i++) {
197  Callback* cb = &L.callbacks[i];
198  if (level >= cb->level) {
199  init_event(&ev, cb->udata);
200  va_start(ev.ap, fmt);
201  cb->fn(&ev);
202  va_end(ev.ap);
203  }
204  }
205 #endif
206 
207  unlock();
208 }
